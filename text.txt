```
CREATE TABLE `statistics` (
	`id` INT NOT NULL AUTO_INCREMENT,
	`id_user` INT NOT NULL DEFAULT '0',
	`id_nps` INT NOT NULL DEFAULT '0',
	`order` JSON NOT NULL,
	`result` INT NOT NULL DEFAULT '0',
	PRIMARY KEY (`id`) USING BTREE,
	INDEX `id_user` (`id_user`) USING BTREE,
	INDEX `id_nps` (`id_nps`) USING BTREE,
	CONSTRAINT `id_user` FOREIGN KEY (`id_user`) REFERENCES `users` (`id`) ON UPDATE NO ACTION ON DELETE NO ACTION,
	CONSTRAINT `id_nps` FOREIGN KEY (`id_nps`) REFERENCES `nps` (`id`) ON UPDATE NO ACTION ON DELETE NO ACTION
)
```
```
CREATE TABLE `ingredients` (
	`id` INT NOT NULL AUTO_INCREMENT,
	`type` VARCHAR(50) NOT NULL COLLATE 'utf8mb4_0900_ai_ci',
	`name` VARCHAR(100) NOT NULL COLLATE 'utf8mb4_0900_ai_ci',
	`color` VARCHAR(11) NOT NULL COLLATE 'utf8mb4_0900_ai_ci',
	PRIMARY KEY (`id`) USING BTREE
)
```
